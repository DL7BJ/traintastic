64inline bool checkMagicData(const Tbase& message)
65{
66 static_assert(std::is_base_of_v<Tbase, T>);
67 static_assert(sizeof(T) == sizeof(Tbase));
68 static_assert(Tbase::dataLen == T::magicData.size());
69 static_assert(Tbase::dataLen == T::magicMask.size());
70 for(uint8_t i = 0; i < Tbase::dataLen; i++)
