142inline std::string buildCommand(std::string_view command, uint16_t objectId, std::initializer_list<std::string_view> options)
143{
144 std::string s(command);
145 s.append("(").append(std::to_string(objectId));
146 for(auto option : options)
147 s.append(", ").append(option);
148 s.append(")\n");
149 return s;
150}
