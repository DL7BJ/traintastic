310 inline std::string dccPacket(std::span<const uint8_t> packet, Track track = Track::Main)
311 {
312 assert(packet.size() <= 5);
313 assert(track == Track::Main || track == Track::Programming);
314 std::string command(track == Track::Programming ? "<P" : "<M");
315 command.append(" 0 "); // register 0
316 command.append(toHex(packet.data(), packet.size(), true));
317 command.append(">\n");
318 return command;
319 }

321 template<class T, std::enable_if_t<std::is_trivially_copyable_v<T> && sizeof(T) <= 5, bool> = true>
322 inline std::string dccPacket(const T& packet, Track track = Track::Main)
323 {
324 return dccPacket(std::span<const uint8_t>(reinterpret_cast<const uint8_t*>(&packet), sizeof(T)), track);
