34inline void pushSet(lua_State* L, const char* setName, lua_Integer value)
35{
36 lua_getglobal(L, setName); // get tabel with all values: key=int, value=set as userdata
37 assert(lua_type(L, -1) == LUA_TTABLE);
38 lua_rawgeti(L, -1, value); // get userdata
39 if(lua_isnil(L, -1)) // value not in table
40 {
41 lua_pop(L, 1); // remove nil
42 *static_cast<lua_Integer*>(lua_newuserdata(L, sizeof(value))) = value;
43 luaL_setmetatable(L, setName);
44 lua_pushvalue(L, -1); // copy set userdata on stack
45 lua_rawseti(L, -3, value); // add set value to table
46 }
47 lua_insert(L, lua_gettop(L) - 1); // swap tabel and userdata
48 lua_pop(L, 1); // remove tabel
49}
